 69 /**
 70  * Parses the command passed to the server
 71  */
 72 Server.parseCommand = function() {
 73     var command = process.argv[2];
 74     if(command === 'webserver') {
 75         return 'webserver';
 76     } else if(command === 'load') {
 77         return 'load';
 78     } else if(command === 'clear') {
 79         return 'clear';
 80     } else {
 81         if(command != null) 
 82             console.log("Error: Unrecognized command "+command);
 83         Server.printUsage();
 84         process.exit(1);
 85     }
 86 };

 88 /**
 89  * Merge command line configuration with the default server options
 90  */
 91 Server.parseOptions = function() {
 92     var options = {};
 93     var lastSwitch = null;
 94     var currentOption = null;
 95     var currentValidator = null;
 96 
 97     for(var i=3; i<process.argv.length; i++) {
 98         var arg = process.argv[i];
 99         if(currentOption != null) {
100             if(currentValidator == null || currentValidator(arg)) {
101                 options[currentOption] = arg;
102                 currentOption = null;
103                 currentValidator = null;
104                 lastSwitch = null;
105             } else {
106                 console.log("Error: Invalid value "+arg+" for options "+lastSwitch+" ["+currentOption+"]");
107                 Server.printUsage();
108                 process.exit(1);
109             }
110         } else if(arg.indexOf("-") == 0) {
111             if(Server.optionSwitches[arg] != null) {
112                 currentOption = Server.optionSwitches[arg];
113                 currentValidator = Server.defaultOptions[currentOption].validate;
114                 lastSwitch = arg;
115             } else {
116                 console.log("Error: Unregognized option "+arg);
117                 Server.printUsage();
118                 process.exit(1);
119             }
120         }
121     }
122     
123     for(var p in Server.defaultOptions) {
124         if(options[p] == null) {
125             options[p] = Server.defaultOptions[p].def;
126         }
127     }
128 
129     return options;
130 };

167         storeOptions['persistent'] = false;
168         storeOptions['treeOrder'] == parseInt(options['store-tree-order']);

170         storeOptions['persistent'] = true;
171         storeOptions['engine'] = 'mongodb';
172 
173         if(options['store-overwrite'] == 'true') {
174             storeOptions['overwrite'] = true;
175         } else {
176             storeOptions['overwrite'] = false;
177         }
178 
179         storeOptions['name'] = options['store-name'];
180         if(options['store-mongo-domain']) {
181             storeOptions['mongoDomain'] = options['store-mongo-domain'];
182         }
183         if(options['store-mongo-port']) {
184             storeOptions['mongoPort'] = parseInt(options['store-mongo-port']);
185         }
186     }
187 
188     new Store.Store(storeOptions,function(store) {
189         if(store!=null) {
190             Server.store = store;
191             callback();
192         } else {
193             console.log("(!!) Error creating RDFStore");
194             process.exit(2);
195         }
196     });

199 /**
200  * Starts the server, parsing the options and handling requests
201  * using the right handler function.
202  */
203 Server.start = function() {
204     var options = Server.parseOptions();
205     var command = Server.parseCommand();
206     
207     if(process.argv.length<3) {
208         Server.printUsage();
209         process.exit(0);
210     } else {
211         if(command === 'webserver') {
212             Server.startStore(options,function(){
213                 if(options['protocol']==='http') {
214                     http.createServer(Server.routeRequest(options)).listen(parseInt(options['port']))
215                 } else if(options['protocol']==='https') {
216                     var httpsOptions = {
217                         key: fs.readFileSync(options['ssl-key']),
218                         cert: fs.readFileSync(options['ssl-cert']),
219                         requestCert: false
220                     };
221                     https.createServer(httpsOptions,Server.routeRequest(options)).listen(parseInt(options['port']))
222                 }
223             });
224         } else if(command === 'clear') {
225             options['store-overwrite'] = 'true';
226             console.log(options);
227             Server.startStore(options,function(){
228                 process.exit(0);
229             });
230         } else if(command === 'load') {
231             Server.startStore(options,function(){
232                 var toLoad = process.argv[3];
233                 var dstGraph = process.argv[4];
234                 if(dstGraph != null && dstGraph[0]==='-')
235                     dstGraph = null;
236 
237                 if(toLoad === 'stdin') {
238                     var mediaType = options['media-type'];
239                     if(mediaType == null) {
240                         var data = "";
241                         process.stdin.resume();
242                         process.stdin.setEncoding('utf8');
243                         process.stdin.on('data', function (chunk) {
244                             data = data + chunk;
245                         });
246 
247                         process.stdin.on('end', function () {
248                             if(dstGraph != null) {
249                                 Server.store.load(mediaType, data, dstGraph, function(){
250                                     process.exit(0);
251                                 });
252                             } else {
253                                 Server.store.load(mediaType, data, function(){
254                                     process.exit(0);
255                                 });
256                             }
257                         });                    
258                     } else {
259                         console.log("Error: Media type used to encode incoming data must be specified using the --media-type flag when loading data from disk or stdin");
260                         process.exit(3);
261                     }
262                 } else {
263                     if(toLoad.indexOf("file:/")==0) {
264                         var mediaType = options['media-type'];
265                         if(mediaType == null) {
266                             if(dstGraph != null) {
267                                 Server.store.load(mediaType, toLoad, dstGraph, function(){
268                                     process.exit(0);
269                                 });
270                             } else {
271                                 Server.store.load(mediaType, toLoad, function(){
272                                     process.exit(0);
273                                 });
274                             }
275                         } else {
276                             console.log("Error: Media type used to encode incoming data must be specified using the --media-type flag when loading data from disk or stdin");
277                             process.exit(3);
278                         }
279                     } else {
280                             if(dstGraph != null) {
281                                 Server.store.load('remote', toLoad, dstGraph, function(){
282                                     process.exit(0);
283                                 });
284                             } else {
285                                 Server.store.load('remote', toLoad, function(){
286                                     process.exit(0);
287                                 });
288                             }
289                     }
290                 }
291             });
292         }
293     }

329 /**
330  * Adds a coercion annotation to a json-ld object
331  *
332  * @arguments
333  * @param obj jsonld Object where the coerced property will be added
334  * @param prpoerty URI of the property to be coerced
335  * @param type coercion type
336  */
337 Server.jsonldCoerce = function(obj, property, type) {
338     if(obj['@context'] == null) {
339         obj['@context'] = {};
340     }
341     if(obj['@context']['@coerce'] == null) {
342         obj['@context']['@coerce'] = {};
343         obj['@context']['@coerce'][type] = property;
344     } else if(typeof(obj['@context']['@coerce'][type]) === 'string' &&
345               obj['@context']['@coerce'][type] != property) {
346         var oldValue = obj['@context']['@coerce'][type];
347         obj['@context']['@coerce'][type] = [oldValue, property];
348     } else if(typeof(obj['@context']['@coerce'][type]) === 'object') {
349         for(var i=0; i<obj['@context']['@coerce'][type].length; i++) {
350             if(obj['@context']['@coerce'][type][i] === property)  {
351                 return obj;
352             }
353         }
354 
355         obj['@context']['@coerce'][type].push(property);
356     } else {
357         obj['@context']['@coerce'][type] = property;
358     }
359 
360     return obj;
361 };

363 /**
364  * Transforms a RDF JS Interfaces API Graph object into a JSON-LD serialization
365  *
366  * @arguments
367  * @param graph JS RDF Interface graph object to be serialized
368  * @param rdf JS RDF Interface RDF environment object
369  */
370 Server.graphToJSONLD = function(graph, rdf) {
371     var nodes = {};
372     
373     graph.forEach(function(triple) {
374         var subject = triple.subject.valueOf();
375         var node = nodes[subject];
376         if(node == null) {
377             node = {"@subject" : subject, "@context": {}};
378             nodes[subject] = node;
379         }
380 
381         var predicate = triple.predicate.valueOf();
382         if(predicate === "http://www.w3.org/1999/02/22-rdf-syntax-ns#type") {
383             predicate = "@type";
384         }
385 
386         var property  = null;
387         var isCURIE = false;
388         property = rdf.prefixes.shrink(predicate);
389 
390         if(property != predicate) {
391             isCURIE = true;
392         }
393         if(property.indexOf("#") != -1) {
394             property = property.split("#")[1];
395         } else {
396             property = property.split("/");
397             property = property[property.length-1];
398         }
399 
400         var object = triple.object.valueOf();
401 
402         if(node[property] != null) {
403             if(!isCURIE) {
404                 if(node["@context"][property] != null || property[0] === '@') {
405                     if(typeof(node[property]) === "object") {
406                         node[property].push(object);
407                     } else {
408                         object = [ node[property], object];
409                         node[property] = object;
410                     }
411                 } else {
412                     property = triple.predicate.valueOf();
413                     if(node[property] == null) {
414                         node[property] = object;
415                     } else {
416                         if(typeof(node[property]) === "object") {
417                             node[property].push(object);
418                         } else {
419                             object = [ node[property], object ];
420                             node[property] = object;
421                         }
422                     }
423 
424                     if(typeof(object) === 'string' &&
425                        (object.indexOf("http://") == 0 || object.indexOf("https://") == 0)) {
426                         Server.jsonldCoerce(node, property, "@iri");
427                     }
428                 }
429             } else {
430                 var prefix = property.split(":")[0];
431                 if(typeof(node[property]) === "object") {
432                     node[property].push(object);
433                 } else {
434                     object = [ node[property], object];
435                     node[property] = object;
436                 }
437             }
438         } else {
439             node[property] = object;
440             if(property[0] != '@') {
441                 if(isCURIE == true) {
442                     // saving prefix
443                     var prefix = property.split(":")[0];
444                     node["@context"][prefix] = rdf.prefixes[prefix];
445                 } else {
446                     // saving whole URI in context
447                     node["@context"][property] = triple.predicate.valueOf();
448                 }
449 
450                 if(typeof(object) === 'string' &&
451                    (object.indexOf("http://") == 0 || object.indexOf("https://") == 0)) {
452                     Server.jsonldCoerce(node, property, "@iri");
453                 }
454                 
455             }
456         }
457     });
458 
459     var results = [];
460     for(var p in nodes) {
461         results.push(nodes[p]);
462     }
463 
464     return results;
465 };

467 /**
468  * Returns a function that route HTTP requests according to the configuration
469  * options passed as arguments
470  *
471  * @arguments
472  * @param {Object} options map of configuration options with parameters passed to the server.
473  */
474 Server.routeRequest = function(options) {
475     var CORSEnabled = options['cors-enabled'] === 'true';
476     
477     // Routes -> handlers mapping
478     var routes = {};
479     routes[options['path']] = "SPARQLProtocolHandler";
480     
481     // routing logic
482     return function(req,res) {
483         try {
484             if(CORSEnabled)
485                 res.withCORSHeader = Server.withCORSHeader;
486 
487             var data = "";
488             req.on('data', function(chunk){
489                 data = data + chunk;
490             });
491 
492             req.on('end', function(){
493                 req.data = data;
494 
495                 var handler = null;
496                 var components = null;
497 
498                 for(var path in routes) {
499                     var handlerString = routes[path];
500                     var requestPath = url.parse(req.url,true,true)['pathname'];
501                     req.requestPath = requestPath;
502 
503                     if(requestPath && requestPath[requestPath.length-1] == '/') {
504                         requestPath = requestPath.substring(0,requestPath.length-1);
505                     }
506 
507                     if(requestPath === path ) {
508                         handler = handlerString;
509                         break;
510                     }
511                 }
512 
513                 if(handler != null) {
514                     Server[handler](req, res, data);
515                 } else {
516                     res.writeHead(404, {"Content-Type":"text/plain"});
517                     res.end();
518                 }
519             });
520 
521         } catch(e) {
522             console.log("(!!) Error: "+e.message);
523             console.log(e.stack);
524             res.withCORSHeader(500,{"Content-Type":"text/plain"});
525             res.end();
526         }
527     };
528 };

530 /**
531  * Handler function that executes a SPARQL query and build a response according
532  * to the SPARQL HTTP Protocol
533  *
534  * @arguments
535  * @param req Node.JS request object
536  * @param res Node.JS response object
537  * @param data textual data send the in the HTTP request
538  */
539 Server.SPARQLProtocolHandler = function(req, res, data) {
540     var params;
541     var accepts = Server.mediaTypes(req);
542 
543     if(req.method === 'GET') {
544         params = url.parse(req.url,true,true)['query'];
545     } else if(req.method === 'POST') {
546         var mediaType = req.headers['content-type'] || req.headers['Content-Type'];
547         if(mediaType == 'application/xml') {
548             // not supported
549             
550         } else {
551             // media type must be application/x-www-form-urlencoded
552             params = querystring.parse(data);
553         }
554     }
555     var query = params['query'];
556     var defaultGraphUris = params['default-graph-uri'] || [];
557     var namedGraphUris = params['named-graph-uri'] || [];
558     if(typeof(namedGraphUris) === 'string')
559         namedGraphUris = [namedGraphUris];
560     if(typeof(defaultGraphUris) === 'string')
561         defaultGraphUris = [defaultGraphUris];
562 
563     try {
564         // Run the query
565         Server.store.execute(query, defaultGraphUris, namedGraphUris, function(success, results) {
566             try {
567                 // Check if this is a RDF JS Interfaces graph or a collection of bindings
568                 if(success) {
569                     if(results.constructor == Array) {
570                         // bindings
571                         Server.buildResponseBindings(accepts, results, res);
572                     } else if(results.constructor == Boolean) {
573                         // ask
574                         Server.buildResponseBoolean(accepts, results, res);
575                     } else {
576                         // construct
577                         Server.buildResponseGraph(accepts, results, res);
578                     }
579                 } else {
580                     if(res.withCORSHeader != null) {
581                         res.withCORSHeader(200, {"Content-Type":"application/rdf+xml"});
582                     } else {
583                         res.writeHead(200,{"Content-Type":"application/rdf+xml"});
584                     }
585                     res.end(new Buffer("Query malformed"), 'utf-8');
586                 }
587             } catch(e) {
588                 console.log("(!!) Error: "+e.message);
589                 console.log(e.stack);
590                 if(res.withCORSHeader != null) {
591                     res.withCORSHeader(500, {"Content-Type":"text/plain"});
592                 } else {
593                     res.writeHead(500,{"Content-Type":"text/plain"});
594                 }
595                 res.end(new Buffer("Query refused"), 'utf-8');
596             }
597         });
598     } catch(e) {
599         console.log("(!!) Error: "+e.message);
600         console.log(e.stack);
601 
602         if(res.withCORSHeader != null) {
603             res.withCORSHeader(500, {"Content-Type":"text/plain"});
604         } else {
605             res.writeHead(500,{"Content-Type":"text/plain"});
606         }
607         res.end(new Buffer("Query refused"), 'utf-8');
608     }
609 };

611 /**
612  * Builds an SPARQL HTTP protocol response for a collection of bindings
613  * returned by the RDF store
614  */
615 Server.buildResponseBindings = function(mediaTypes, bindings, res) {
616     var accepts;
617 
618     for(var i=0; i<mediaTypes.length; i++) {
619         if(mediaTypes[i] === 'application/json') {
620             accepts = mediaTypes[i];
621             break;
622         }
623     } 
624     if(accepts == null) {
625         // by default xml
626         accepts = 'application/rdf+xml';
627     };
628 
629     if(accepts === 'application/json') {
630         var varNames = {};
631         var genBindings = [];
632         for(var i=0; i<bindings.length; i++) {
633             var result = bindings[i];
634             for(var p in bindings[i]) {
635                 varNames[p] = true;
636             }
637         }
638         var head = {'variables':[]};
639         for(var p in varNames) {
640             head['variables'].push({'name':p});
641         }
642 
643         if(res.withCORSHeader != null) {
644             res.withCORSHeader(200, {"Content-Type":"application/json"});
645         } else {
646             res.writeHead(200,{"Content-Type":"application/json"});
647         }
648         res.end(new Buffer(JSON.stringify({'head':head,'results':bindings})), 'utf-8');
649     } else {
650         var response = '<?xml version="1.0" encoding="UTF-8"?><sparql xmlns="http://www.w3.org/2005/sparql-results#">';
651         var results = '<results>';
652 
653         var varNames = {};
654         for(var i=0; i<bindings.length; i++) {               
655             var nextResult = '<result>';
656 
657             var result = bindings[i];
658             
659             for(var p in result) {
660                 varNames[p] = true;
661                 nextResult = nextResult+'<binding name="'+Server.xmlEncode(p)+'">';
662                 if(result[p].token === 'uri') {
663                     nextResult = nextResult+"<uri>"+result[p].value+"</uri>";
664                 } else if(result[p].token === 'literal') {
665                     nextResult = nextResult+"<literal";
666                     if(result[p].lang != null ) {
667                         nextResult = nextResult + ' xml:lang="'+result[p].lang+'" ';
668                     }
669                     if(result[p].type != null ) {
670                         nextResult = nextResult + ' datatype="'+result[p].type+'" ';
671                     }
672                     nextResult = nextResult+">"+Server.xmlEncode(result[p].value)+"</literal>";
673                 } else {
674                     nextResult = nextResult+"<bnode>"+result[p].value+"</bnode>";
675                 }
676                 nextResult + nextResult+'</binding>';
677             }
678 
679             nextResult = nextResult+'</result>';
680             results = results+nextResult;
681         }
682         results = results + '</results>';
683 
684         var head = '<head>'
685         for(var varName in varNames) {
686             head = head + '<variable name="'+Server.xmlEncode(varName)+'"/>';
687         }
688         head = head + '</head>';
689

694         } else {
695             res.writeHead(200,{"Content-Type":"application/sparql-results+xml"});
696         }
697         res.end(new Buffer(response), 'utf-8');

738 /**
739  * Builds an SPARQL HTTP protocol response for RDF JS interface graph
740  * returned by the RDF store
741  */
742 Server.buildResponseGraph = function(mediaTypes, graph, res) {
743     var accepts;
744 
745     for(var i=0; i<mediaTypes.length; i++) {
746         if(mediaTypes[i] === 'application/json') {
747             accepts = mediaTypes[i];
748             break;
749         } else if(mediaTypes[i] === 'text/turtle' || mediaTypes[i] === 'text/ttl' || mediaTypes[i] === 'text/n3' || mediaTypes[i] === 'text/plain') {
750             accepts = mediaTypes[i];
751             break;
752         }
753     } 
754     if(accepts == null) {
755         // by default xml
756         accepts = 'application/rdf+xml';
757     };
758 
759     if(accepts === 'application/json') {
760         var jsonld = Server.graphToJSONLD(graph,Server.store.rdf);
761         if(res.withCORSHeader != null) {
762             res.withCORSHeader(200, {"Content-Type":"application/json"});
763         } else {
764             res.writeHead(200,{"Content-Type":"application/json"});
765         }
766         res.end(new Buffer(JSON.stringify(jsonld)), 'utf-8');
767     } else if(accepts === 'text/turtle' || accepts === 'text/ttl' || accepts === 'text/n3' || accepts === 'text/plain') {
768         var text = graph.toNT();
769         if(res.withCORSHeader != null) {
770             res.withCORSHeader(200, {"Content-Type":accepts});
771         } else {
772             res.writeHead(200,{"Content-Type":accepts});
773         }
774         res.end(new Buffer(text), 'utf-8');
775     } else {
776         var ns = {};
777         var nsCounter = 0;
778         var triples = graph.toArray();
779         var triple, predicate, parts, prefix, suffix;
780         var subject, xmlPredicate, tripleXML;
781 
782         var acum  = "";
783 
784         for(var i=0; i<triples.length; i++) {
785             triple = triples[i];
786 
787             if(triple.subject.interfaceName === 'BlankNode') {
788                 tripleXML = '<rdf:Description rdf:nodeID="blank'+triple.subject.bnodeId;
789                 tripleXML = tripleXML+'">';
790             } else {
791                 tripleXML = '<rdf:Description rdf:about="'+triple.subject.valueOf();
792                 tripleXML = tripleXML+'">';
793             }
794 
795             predicate = triple.predicate.valueOf();
796             if(predicate.indexOf("#") != -1) {
797                 parts = predicate.split("#");
798                 prefix = parts[0]+"#";
799                 suffix = parts[1];
800             } else {
801                 parts = predicate.split("/");
802                 suffix = parts.pop();
803                 prefix = parts.join("/")+"/";
804             }
805      
806             if(ns[prefix] == null) {
807                 xmlPredicate = "ns"+nsCounter;
808                 nsCounter++;
809                 ns[prefix] = xmlPredicate;
810             } else {
811                 xmlPredicate = ns[prefix];
812             }
813             xmlPredicate = xmlPredicate+":"+suffix;
814 
815             
816             if(triple.object.interfaceName === "Literal") {
817                 if(triple.object.datatype != null && triple.object.language != null) {
818                     tripleXML = tripleXML + '<'+xmlPredicate+'>'+Server.xmlEncode(triple.object.nominalValue)+'</'+xmlPredicate+'>';
819                 } else if(triple.object.datatype != null) {
820                     tripleXML = tripleXML + '<'+xmlPredicate+' rdf:datatype="'+triple.object.datatype+'">'+Server.xmlEncode(triple.object.nominalValue)+'</'+xmlPredicate+'>';                    
821                 } else {
822                     tripleXML = tripleXML + '<'+xmlPredicate+' xml:lang="'+triple.object.language+'">'+Server.xmlEncode(triple.object.nominalValue)+'</'+xmlPredicate+'>';                    
823                 }
824             } else if(triple.object.interfaceName === 'NamedNode') {
825                     tripleXML = tripleXML + '<'+xmlPredicate+' rdf:resource="'+triple.object.valueOf()+'"/>';
826             } else if(triple.object.interfaceName === 'BlankNode') {
827                     tripleXML = tripleXML + '<'+xmlPredicate+' rdf:nodeID="blank'+triple.object.bnodeId+'"/>';
828             }
829 
830             tripleXML = tripleXML + "</rdf:Description>";
831             acum = acum + tripleXML;
832         }
833 
834         var xml = '<?xml version="1.0"  encoding="UTF-8"?>';
835         xml = xml + '<rdf:RDF xmlns:rdf="http://www.w3.org/1999/02/22-rdf-syntax-ns#"';
836         for(var prefix in ns) {
837             var prefixNs = ns[prefix];
838             xml = xml + ' xmlns:'+prefixNs+'="'+prefix+'"';
839         }
840         xml = xml+'>'+acum+'</rdf:RDF>';
841 
842         if(res.withCORSHeader != null) {
843             res.withCORSHeader(200, {"Content-Type":"application/rdf+xml"});
844         } else {
845             res.writeHead(200,{"Content-Type":"application/rdf+xml"});
846         }
847         res.end(new Buffer(xml), 'utf-8');
848     }
849 };
